-- Triggers
 create table emp (eid int primary key auto_increment,ename varchar(20),salary int);

 select * from emp;
 -- message trigger
 
 create table emp_mss (message varchar(20));
 
 -- trigger message
 delimiter //
 create trigger emp_insert_trigger
 after insert on emp 
 for each row begin insert into emp_mss(message)values(concat('emp added: ',NEW.ename));
 end//
 delimiter ;

  create trigger emp_update_trigger
  after update on emp

  concat('emp salary changed : ',old.salary,'to',new.salary));
